# HQ-SRV1 — DMZ Network Configuration
# Server moved from VLAN 400 (HQ-DSW1) to DMZ (HQ-FW1 Gi0/2)
# New IP: 10.1.40.10/24 | Gateway: 10.1.40.1 (HQ-FW1 DMZ interface)

# Flush old IP and set new DMZ address
ip addr flush dev eth0
ip address add 10.1.40.10/24 dev eth0
ip link set dev eth0 up
ip route replace default via 10.1.40.1

# Verify
ip addr show eth0
ip route show
