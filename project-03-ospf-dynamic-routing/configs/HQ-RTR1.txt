! ============================================================
! DEVICE: HQ-RTR1
! PROJECT: 03 — OSPF Dynamic Routing
! ROLE: ABR between Area 0 (WAN backbone) and Area 1 (HQ campus)
! ROUTER ID: 10.0.255.1
! BUILD DATE: 2026-04-12
! ============================================================

hostname HQ-RTR1

! --- Loopback (stable router ID anchor) ---
interface Loopback0
 description ROUTER-ID-LOOPBACK
 ip address 10.0.255.1 255.255.255.255

! --- LAN interface (trunk to HQ-DSW1) ---
interface Ethernet0/0
 description TRUNK-TO-HQ-DSW1-E0/0
 no ip address
 no shutdown

interface Ethernet0/0.100
 description GATEWAY-VLAN100-ENGINEERING
 encapsulation dot1Q 100
 ip address 10.1.100.1 255.255.255.0
 ip helper-address 10.1.99.50
 ipv6 address 2001:DB8:1:100::1/64
 ipv6 enable
 ipv6 ospf 10 area 1

interface Ethernet0/0.200
 description GATEWAY-VLAN200-SALES
 encapsulation dot1Q 200
 ip address 10.1.200.1 255.255.255.0
 ip helper-address 10.1.99.50

interface Ethernet0/0.300
 description GATEWAY-VLAN300-GUEST
 encapsulation dot1Q 300
 ip address 10.1.44.1 255.255.255.0
 ip helper-address 10.1.99.50

interface Ethernet0/0.400
 description GATEWAY-VLAN400-SERVERS
 encapsulation dot1Q 400
 ip address 10.1.40.1 255.255.255.0

interface Ethernet0/0.999
 description GATEWAY-VLAN999-MANAGEMENT
 encapsulation dot1Q 999
 ip address 10.1.99.1 255.255.255.0

! --- Primary WAN link to BR-RTR1 (BACKUP path — cost 100) ---
interface Ethernet0/1
 description WAN-TO-BR-RTR1-E0/1
 bandwidth 1000
 delay 1000
 ip address 10.0.0.1 255.255.255.252
 ip ospf authentication message-digest
 ip ospf message-digest-key 1 md5 OSPF-WAN-KEY
 ip ospf network point-to-point
 ip ospf cost 100
 ip ospf bfd
 bfd interval 300 min_rx 300 multiplier 3
 ipv6 address 2001:DB8:0:1::1/126
 ipv6 enable
 ipv6 ospf 10 area 0
 ipv6 ospf network point-to-point
 no shutdown

! --- WAN link to WAN-RTR1 (PREFERRED path — cost 10) ---
interface Ethernet0/2
 description WAN-TO-WAN-RTR1-E0/1
 ip address 10.0.0.5 255.255.255.252
 ip ospf authentication message-digest
 ip ospf message-digest-key 1 md5 OSPF-WAN-KEY
 ip ospf network point-to-point
 ip ospf cost 10
 ip ospf bfd
 bfd interval 300 min_rx 300 multiplier 3
 no shutdown

! --- OSPF Process (IPv4) ---
router ospf 1
 router-id 10.0.255.1
 log-adjacency-changes detail
 passive-interface default
 no passive-interface Ethernet0/1
 no passive-interface Ethernet0/2
 ! WAN backbone links in Area 0
 network 10.0.255.1 0.0.0.0 area 0
 network 10.0.0.0 0.0.0.3 area 0
 network 10.0.0.4 0.0.0.3 area 0
 ! HQ campus VLANs in Area 1
 network 10.1.100.0 0.0.0.255 area 1
 network 10.1.200.0 0.0.0.255 area 1
 network 10.1.44.0 0.0.0.255 area 1
 network 10.1.40.0 0.0.0.255 area 1
 network 10.1.99.0 0.0.0.255 area 1
 ! Summarize Area 1 into Area 0 at the ABR
 area 1 range 10.1.0.0 255.255.0.0

! --- OSPFv3 Process (IPv6) ---
ipv6 unicast-routing
ipv6 router ospf 10
 router-id 10.0.255.1

! --- IP SLA — probe directly connected BR-RTR1 E0/1 IP ---
! WHY: Target must be reachable WITHOUT OSPF (directly connected /30)
!      so the probe survives an OSPF failure and keeps the track UP.
ip sla 10
 icmp-echo 10.0.0.2 source-ip 10.0.0.1
 timeout 1000
 threshold 1000
 frequency 5
ip sla schedule 10 life forever start-time now

! --- Object tracking ---
track 10 ip sla 10 reachability

! --- Floating static backup (AD 250 — last resort, OSPF preferred) ---
ip route 10.2.0.0 255.255.0.0 10.0.0.2 250 track 10

! --- No static routes (all replaced by OSPF in Project 03) ---

end
